559 void mouseCallback(void) { LOG
560  static unsigned char packet[] = { 0, 0, 0 };
561  static short counter = 0;
562 
563  unsigned long datatemp;
564 
565  if (mouse_read(&datatemp) != 0) {
566  printf("mouseCallback: mouse_read failed.\n");
567  }
568 
569  packet[counter] = datatemp;
570 
571  if (counter == 0 && !bit_isset(packet[0], 3))
572  return;
573 
574  counter = (counter + 1) % 3;
575 
576  if (counter != 0)
577  return;
578 
579  mouse_state.ldown = !!bit_isset(packet[0], 0); // LB
580  mouse_state.rdown = !!bit_isset(packet[0], 1); // RB
581  mouse_state.mdown = !!bit_isset(packet[0], 2); // MB
582  mouse_state.diffx = !bit_isset(packet[0], 4) ? packet[1] : (char)(packet[1]);
583  mouse_state.diffy = !bit_isset(packet[0], 5) ? packet[2] : (char)(packet[2]);
584  mouse_state.ovfx = !!bit_isset(packet[0], 6); // XOV
585  mouse_state.ovfy = !!bit_isset(packet[0], 7); // YOV
586 
587  printf("B1=0x%02X B2=0x%02X B3=0x%02X LB=%d MB=%d RB=%d XOV=%d YOV=%d X=%04d Y=%04d\n",
588  packet[0], // B1
589  packet[1], // B2
590  packet[2], // B3
591  mouse_state.ldown, // LB
592  mouse_state.mdown, // MB
593  mouse_state.rdown, // RB
594  mouse_state.ovfx, // XOV
595  mouse_state.ovfy, // YOV
596  mouse_state.diffx, // X
597  mouse_state.diffy); // Y
598 
599  mouse_state.updated = 1;
600 }

785 int window_remove_tab(window_t* window, int tab_num) {
786  if (tab_num < 0 || tab_num >= TAB_COUNT || tab_num == TAB_CONSOLE) {
787  window_set_log_message(window, "Tab Manager: Invalid tab.");
788  return 0;
789  }
790 
791  if (!window->tabs[tab_num]) {
792  window_set_log_message(window, "Tab Manager: Tab doesn't exist");
793  return 0;
794  }
795 
796  tab_destroy(window->tabs[tab_num]);
797  window->tabs[tab_num] = NULL;
798  if (window->prev_current_tab == tab_num) {
799  window->prev_current_tab = TAB_CONSOLE;
800  }
801  if (window->current_tab == tab_num) {
802  window->current_tab = TAB_CONSOLE;
803  }
804  window_set_log_message(window, "Tab Manager: Tab closed with success.");
805 
806  return 1;
807 }
